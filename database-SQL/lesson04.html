DML - Data Manipulation Language

Linguagem para manipulação de dados.

INSERT


                            // SEM especificação das colunas (tem que seguir a ordem)
                                INSERT INTO user
                                VALUES (null, 'Heviane Bastos da Costa', 'heviane@gmail.com', '123456', NOW(), NOW());

                            // COM especificação das colunas
                                INSERT INTO user (name, email, password, created_date, updated_date)
                                VALUES ('Heviane Bastos da Costa', 'heviane@gmail.com', '123456', NOW(), NOW());

UPDATE


                            // ATUALIZA todos os registros
                                UPDATE user
                                    SET password = '999999'
    
                            // ATUALIZA apenas um registro
                                UPDATE user
                                    SET email= 'maria@teste.com'
                                    WHERE id = 2;

DELETE


                            DELETE FROM table_name
                            WHERE some_column = some_value;
